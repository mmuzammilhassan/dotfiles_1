--this is .lua code for popup on the top-right-corner for neovim
local NuiPopup = require("nui.popup")
local popup = NuiPopup({
  position = {
    row = 1,
    col = vim.o.columns - 42, -- right side
  },
  size = {
    width = 40,
    height = 6,
  },
  enter = false,
  focusable = false,
  border = {
    style = "rounded",
  },
})

popup:mount()

vim.api.nvim_buf_set_lines(popup.bufnr, 0, 1, false, {
  "ðŸ¦‹ Top-right corner popup",
  "Popup works inside Tmux âœ…",
})

-- auto-close after 3 seconds
vim.defer_fn(function()
  popup:unmount()
end, 3000)
